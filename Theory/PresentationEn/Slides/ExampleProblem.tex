\begin{column}[T]{0.35\textwidth}
\begin{block}{The minimization problem}
$$
x_1^2+(x_2-2)^2\rightarrow \min.
$$
Restrictions:
$$
\left\{
\begin{array}{l}
x_1^2+x_2^2-9; \\
x_1^2+(x_2-4)^2-9.
\end{array}
\right.
$$
\end{block}
\end{column}
