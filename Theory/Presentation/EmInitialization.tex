\begin{frame}{Стартовые условия для МЭ}

\begin{block}{Задан:}
Коэффициент $\alpha$ такой, что $\alpha+1/\alpha<2\sqrt[n]{\alpha}$.
\end{block}

\begin{block}{Инициализация:}
\begin{enumerate}
\item Выбрать стартовую точку $x_0\in E_n$ и начальный радиус $r_0$, такой что $||x_0-x^*||\le r_0$.
\item Положить $B_0:=E$, где $E$ -- единичная матрица.
\end{enumerate}
\end{block}

\begin{exampleblock}{}
Перейти к очередной итерации со значениями $x_0$, $r_0$, $B_0$.
\end{exampleblock}

\end{frame}
