\begin{frame}{О сходимости метода эллипсоидов}

\begin{rustheorem}[Локализация $x^*$ в эллипсоиде]
Генерируемая методом эллипсоидов последовательность точек $\{x_k\}_{k=0}^\infty$ удовлетворяет неравенствам $$||B_k^{-1}(x_k-x^*)||\le r_k,\quad k=0,1,2,\ldots$$
\end{rustheorem}

\begin{ruscorollary}
Эллипсоид $\varepsilon_k=\{x:||B_k^{-1}(x_k-x)||\le r_k\}$ содержит точку $x^*$.
\end{ruscorollary}

\end{frame}
